 SELECT Title FROM albums

   ...> WHERE Name = 'AC/DC';


Récupérer la liste des titres de l'album "Let There Be Rock" : 

 SELECT Name FROM tracks
   ...> JOIN albums ON tracks.AlbumId = albums.AlbumId
   ...> WHERE Title = "Let There Be Rock";
